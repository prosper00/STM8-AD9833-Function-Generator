203 {
204  /*----------------------*/
205  /* Check the parameters */
206  /*----------------------*/
207 
208  assert_param(IS_GPIO_MODE_OK(GPIO_Mode));
209  assert_param(IS_GPIO_PIN_OK(GPIO_Pin));
210 
211  /* Reset corresponding bit to GPIO_Pin in CR2 register */
212  GPIOx->CR2 &= (uint8_t) (~(GPIO_Pin));
213 
214  /*-----------------------------*/
215  /* Input/Output mode selection */
216  /*-----------------------------*/
217 
218  if ((((uint8_t) (GPIO_Mode)) & (uint8_t) 0x80) != (uint8_t) 0x00) /* Output mode */
219  {
220  if ((((uint8_t) (GPIO_Mode)) & (uint8_t) 0x10) != (uint8_t) 0x00) /* High level */
221  {
222  GPIOx->ODR |= (uint8_t) GPIO_Pin;
223  } else /* Low level */
224  {
225  GPIOx->ODR &= (uint8_t) (~(GPIO_Pin));
226  }
227  /* Set Output mode */
228  GPIOx->DDR |= (uint8_t) GPIO_Pin;
229  } else /* Input mode */
230  {
231  /* Set Input mode */
232  GPIOx->DDR &= (uint8_t) (~(GPIO_Pin));
233  }
234 
235  /*------------------------------------------------------------------------*/
236  /* Pull-Up/Float (Input) or Push-Pull/Open-Drain (Output) modes selection */
237  /*------------------------------------------------------------------------*/
238 
239  if ((((uint8_t) (GPIO_Mode)) & (uint8_t) 0x40) != (uint8_t) 0x00) /* Pull-Up or Push-Pull */
240  {
241  GPIOx->CR1 |= (uint8_t) GPIO_Pin;
242  } else /* Float or Open-Drain */
243  {
244  GPIOx->CR1 &= (uint8_t) (~(GPIO_Pin));
245  }
246 
247  /*-----------------------------------------------------*/
248  /* Interrupt (Input) or Slope (Output) modes selection */
249  /*-----------------------------------------------------*/

251  if ((((uint8_t) (GPIO_Mode)) & (uint8_t) 0x20) != (uint8_t) 0x00) /* Interrupt or Slow slope */
