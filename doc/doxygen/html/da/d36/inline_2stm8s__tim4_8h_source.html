481 
482  /* Check the parameters */

486 
487  itenable = (uint8_t) (TIM4->IER & (uint8_t) TIM4_IT);
488 
489  if ((itstatus != (uint8_t) RESET) && (itenable != (uint8_t) RESET)) {
490  bitstatus = (ITStatus) SET;

492  bitstatus = (ITStatus) RESET;
