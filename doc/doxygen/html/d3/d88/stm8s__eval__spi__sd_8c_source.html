714 static SD_CmdAnswer_typedef SD_SendCmd(uint8_t Cmd, uint32_t Arg, uint8_t Crc, uint8_t Answer)
715 {
716  uint32_t i = 0x00;
717  SD_CmdAnswer_typedef retr = {0xFF, 0xFF , 0xFF, 0xFF, 0xFF};
718  uint8_t Frame[6];
719 
720  Frame[0] = (Cmd | 0x40);
722  Frame[1] = (uint8_t)(Arg >> 24);
724  Frame[2] = (uint8_t)(Arg >> 16);
726  Frame[3] = (uint8_t)(Arg >> 8);
728  Frame[4] = (uint8_t)(Arg);
730  Frame[5] = (Crc | 0x01);
732  SD_CS_LOW();
733 
734  for (i = 0; i < 6; i++)
735  {
736  SD_WriteByte(Frame[i]);
737  }
738 
739 switch(Answer)
740  {
741  case SD_ANSWER_R1_EXPECTED :
742  retr.r1 = SD_ReadData();
743  break;
744  case SD_ANSWER_R1B_EXPECTED :
745  retr.r1 = SD_ReadData();
746  retr.r2 = SD_WriteByte(SD_DUMMY_BYTE);
747  /* Set CS High */
748  SD_CS_HIGH();
749 
750  /* Set CS Low */
751  SD_CS_LOW();
752 
753  /* Wait IO line return 0xFF */
754  while (SD_WriteByte(SD_DUMMY_BYTE) != 0xFF);
755  break;
756  case SD_ANSWER_R2_EXPECTED :
757  retr.r1 = SD_ReadData();
758  retr.r2 = SD_WriteByte(SD_DUMMY_BYTE);
759  break;
760  case SD_ANSWER_R3_EXPECTED :
761  case SD_ANSWER_R7_EXPECTED :
762  retr.r1 = SD_ReadData();
763  retr.r2 = SD_WriteByte(SD_DUMMY_BYTE);
764  retr.r3 = SD_WriteByte(SD_DUMMY_BYTE);
765  retr.r4 = SD_WriteByte(SD_DUMMY_BYTE);
766  retr.r5 = SD_WriteByte(SD_DUMMY_BYTE);
767  break;
768  default :
769  break;
770  }
771  return retr;
772 }
