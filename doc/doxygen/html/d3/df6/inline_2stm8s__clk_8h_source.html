649 inline ErrorStatus
650 CLK_ClockSwitchConfig(CLK_SwitchMode_TypeDef CLK_SwitchMode, CLK_Source_TypeDef CLK_NewClock, FunctionalState ITState,
651  CLK_CurrentClockState_TypeDef CLK_CurrentClockState)
652 {
653  CLK_Source_TypeDef clock_master;
654  uint16_t DownCounter = CLK_TIMEOUT;
655  ErrorStatus Swif = ERROR;
656 
657  /* Check the parameters */
658  assert_param(IS_CLK_SOURCE_OK(CLK_NewClock));
659  assert_param(IS_CLK_SWITCHMODE_OK(CLK_SwitchMode));
660  assert_param(IS_FUNCTIONALSTATE_OK(ITState));
661  assert_param(IS_CLK_CURRENTCLOCKSTATE_OK(CLK_CurrentClockState));
662 
663  /* Current clock master saving */
664  clock_master = (CLK_Source_TypeDef) CLK->CMSR;
665 
666  /* Automatic switch mode management */
667  if (CLK_SwitchMode == CLK_SWITCHMODE_AUTO) {
668  /* Enables Clock switch */
669  CLK->SWCR |= CLK_SWCR_SWEN;
670 
671  /* Enables or Disables Switch interrupt */
672  if (ITState != DISABLE) {
673  CLK->SWCR |= CLK_SWCR_SWIEN;
674  } else {
675  CLK->SWCR &= (uint8_t) (~CLK_SWCR_SWIEN);
676  }
677 
678  /* Selection of the target clock source */
679  CLK->SWR = (uint8_t) CLK_NewClock;
680 
681  /* Wait until the target clock source is ready */
682  while ((((CLK->SWCR & CLK_SWCR_SWBSY) != 0) && (DownCounter != 0))) {
683  DownCounter--;
684  }
685 
686  if (DownCounter != 0) {
687  Swif = SUCCESS;
688  } else {
689  Swif = ERROR;
690  }
691  } else /* CLK_SwitchMode == CLK_SWITCHMODE_MANUAL */
692  {
693  /* Enables or Disables Switch interrupt if required */
694  if (ITState != DISABLE) {
695  CLK->SWCR |= CLK_SWCR_SWIEN;
696  } else {
697  CLK->SWCR &= (uint8_t) (~CLK_SWCR_SWIEN);
698  }
699 
700  /* Selection of the target clock source */
701  CLK->SWR = (uint8_t) CLK_NewClock;
702 
703  /* Wait until the target clock source is ready */
704  while ((((CLK->SWCR & CLK_SWCR_SWIF) != 0) && (DownCounter != 0))) {
705  DownCounter--;
706  }
707 
708  if (DownCounter != 0) {
709  /* Enables Clock switch */
710  CLK->SWCR |= CLK_SWCR_SWEN;
711  Swif = SUCCESS;
712  } else {
713  Swif = ERROR;
714  }
715  }
716  if (Swif != ERROR) {
717  /* Switch OFF current clock if required */
718  if ((CLK_CurrentClockState == CLK_CURRENTCLOCKSTATE_DISABLE) && (clock_master == CLK_SOURCE_HSI)) {
719  CLK->ICKR &= (uint8_t) (~CLK_ICKR_HSIEN);
720  } else if ((CLK_CurrentClockState == CLK_CURRENTCLOCKSTATE_DISABLE) && (clock_master == CLK_SOURCE_LSI)) {
721  CLK->ICKR &= (uint8_t) (~CLK_ICKR_LSIEN);
722  } else if ((CLK_CurrentClockState == CLK_CURRENTCLOCKSTATE_DISABLE) && (clock_master == CLK_SOURCE_HSE)) {
723  CLK->ECKR &= (uint8_t) (~CLK_ECKR_HSEEN);
724  }
725  }
726  return (Swif);
727 }
