733 inline void I2C_Init(uint32_t OutputClockFrequencyHz, uint16_t OwnAddress,
734  I2C_DutyCycle_TypeDef I2C_DutyCycle, I2C_Ack_TypeDef Ack,
735  I2C_AddMode_TypeDef AddMode, uint8_t InputClockFrequencyMHz)
736 {
737  uint16_t result = 0x0004;
738  uint16_t tmpval = 0;
739  uint8_t tmpccrh = 0;
740 
741  /* Check the parameters */
742  assert_param(IS_I2C_ACK_OK(Ack));
743  assert_param(IS_I2C_ADDMODE_OK(AddMode));
744  assert_param(IS_I2C_OWN_ADDRESS_OK(OwnAddress));
745  assert_param(IS_I2C_DUTYCYCLE_OK(I2C_DutyCycle));
746  assert_param(IS_I2C_INPUT_CLOCK_FREQ_OK(InputClockFrequencyMHz));
747  assert_param(IS_I2C_OUTPUT_CLOCK_FREQ_OK(OutputClockFrequencyHz));
748 
749 
750  /*------------------------- I2C FREQ Configuration ------------------------*/
751  /* Clear frequency bits */
752  I2C->FREQR &= (uint8_t) (~I2C_FREQR_FREQ);
753  /* Write new value */
754  I2C->FREQR |= InputClockFrequencyMHz;
755 
756  /*--------------------------- I2C CCR Configuration ------------------------*/
757  /* Disable I2C to configure TRISER */
758  I2C->CR1 &= (uint8_t) (~I2C_CR1_PE);
759 
760  /* Clear CCRH & CCRL */
761  I2C->CCRH &= (uint8_t) (~(I2C_CCRH_FS | I2C_CCRH_DUTY | I2C_CCRH_CCR));
762  I2C->CCRL &= (uint8_t) (~I2C_CCRL_CCR);
763 
764  /* Detect Fast or Standard mode depending on the Output clock frequency selected */
765  if (OutputClockFrequencyHz > I2C_MAX_STANDARD_FREQ) /* FAST MODE */
766  {
767  /* Set F/S bit for fast mode */
768  tmpccrh = I2C_CCRH_FS;
769 
770  if (I2C_DutyCycle == I2C_DUTYCYCLE_2) {
771  /* Fast mode speed calculate: Tlow/Thigh = 2 */
772  result = (uint16_t) ((InputClockFrequencyMHz * 1000000) / (OutputClockFrequencyHz * 3));
773  } else /* I2C_DUTYCYCLE_16_9 */
774  {
775  /* Fast mode speed calculate: Tlow/Thigh = 16/9 */
776  result = (uint16_t) ((InputClockFrequencyMHz * 1000000) / (OutputClockFrequencyHz * 25));
777  /* Set DUTY bit */
778  tmpccrh |= I2C_CCRH_DUTY;
779  }
780 
781  /* Verify and correct CCR value if below minimum value */
782  if (result < (uint16_t) 0x01) {
783  /* Set the minimum allowed value */
784  result = (uint16_t) 0x0001;
785  }
786 
787  /* Set Maximum Rise Time: 300ns max in Fast Mode
788  = [300ns/(1/InputClockFrequencyMHz.10e6)]+1
789  = [(InputClockFrequencyMHz * 3)/10]+1 */
790  tmpval = ((InputClockFrequencyMHz * 3) / 10) + 1;
791  I2C->TRISER = (uint8_t) tmpval;
792 
793  } else /* STANDARD MODE */
794  {
795 
796  /* Calculate standard mode speed */
797  result = (uint16_t) ((InputClockFrequencyMHz * 1000000) / (OutputClockFrequencyHz << (uint8_t) 1));
798 
799  /* Verify and correct CCR value if below minimum value */
800  if (result < (uint16_t) 0x0004) {
801  /* Set the minimum allowed value */
802  result = (uint16_t) 0x0004;
803  }
804 
805  /* Set Maximum Rise Time: 1000ns max in Standard Mode
806  = [1000ns/(1/InputClockFrequencyMHz.10e6)]+1
807  = InputClockFrequencyMHz+1 */
808  I2C->TRISER = (uint8_t) (InputClockFrequencyMHz + (uint8_t) 1);
809 
810  }
811 
812  /* Write CCR with new calculated value */
813  I2C->CCRL = (uint8_t) result;
814  I2C->CCRH = (uint8_t) ((uint8_t) ((uint8_t) (result >> 8) & I2C_CCRH_CCR) | tmpccrh);
815 
816  /* Enable I2C */
817  I2C->CR1 |= I2C_CR1_PE;
818 
819  /* Configure I2C acknowledgement */
820  I2C_AcknowledgeConfig(Ack);
821 
822  /*--------------------------- I2C OAR Configuration ------------------------*/
823  I2C->OARL = (uint8_t) (OwnAddress);
824  I2C->OARH = (uint8_t) ((uint8_t) (AddMode | I2C_OARH_ADDCONF) |
825  (uint8_t) ((OwnAddress & (uint16_t) 0x0300) >> (uint8_t) 7));
826 }
