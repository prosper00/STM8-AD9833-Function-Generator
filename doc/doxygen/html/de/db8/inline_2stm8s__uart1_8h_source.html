503 inline void UART1_Init(uint32_t BaudRate, UART1_WordLength_TypeDef WordLength,
504  UART1_StopBits_TypeDef StopBits, UART1_Parity_TypeDef Parity,
505  UART1_SyncMode_TypeDef SyncMode, UART1_Mode_TypeDef Mode)
506 {
507  uint32_t BaudRate_Mantissa = 0, BaudRate_Mantissa100 = 0;
508 
509  /* Check the parameters */
510  assert_param(IS_UART1_BAUDRATE_OK(BaudRate));
511  assert_param(IS_UART1_WORDLENGTH_OK(WordLength));
512  assert_param(IS_UART1_STOPBITS_OK(StopBits));
513  assert_param(IS_UART1_PARITY_OK(Parity));
514  assert_param(IS_UART1_MODE_OK((uint8_t) Mode));
515  assert_param(IS_UART1_SYNCMODE_OK((uint8_t) SyncMode));
516 
517  /* Clear the word length bit */
518  UART1->CR1 &= (uint8_t) (~UART1_CR1_M);
519 
520  /* Set the word length bit according to UART1_WordLength value */
521  UART1->CR1 |= (uint8_t) WordLength;
522 
523  /* Clear the STOP bits */
524  UART1->CR3 &= (uint8_t) (~UART1_CR3_STOP);
525  /* Set the STOP bits number according to UART1_StopBits value */
526  UART1->CR3 |= (uint8_t) StopBits;
527 
528  /* Clear the Parity Control bit */
529  UART1->CR1 &= (uint8_t) (~(UART1_CR1_PCEN | UART1_CR1_PS));
530  /* Set the Parity Control bit to UART1_Parity value */
531  UART1->CR1 |= (uint8_t) Parity;
532 
533  /* Clear the LSB mantissa of UART1DIV */
534  UART1->BRR1 &= (uint8_t) (~UART1_BRR1_DIVM);
535  /* Clear the MSB mantissa of UART1DIV */
536  UART1->BRR2 &= (uint8_t) (~UART1_BRR2_DIVM);
537  /* Clear the Fraction bits of UART1DIV */
538  UART1->BRR2 &= (uint8_t) (~UART1_BRR2_DIVF);
539 
540  /* Set the UART1 BaudRates in BRR1 and BRR2 registers according to UART1_BaudRate value */
541  BaudRate_Mantissa = ((uint32_t) CLK_GetClockFreq() / (BaudRate << 4));
542  BaudRate_Mantissa100 = (((uint32_t) CLK_GetClockFreq() * 100) / (BaudRate << 4));
543  /* Set the fraction of UART1DIV */
544  UART1->BRR2 |= (uint8_t) ((uint8_t) (((BaudRate_Mantissa100 - (BaudRate_Mantissa * 100)) << 4) / 100) &
545  (uint8_t) 0x0F);
546  /* Set the MSB mantissa of UART1DIV */
547  UART1->BRR2 |= (uint8_t) ((BaudRate_Mantissa >> 4) & (uint8_t) 0xF0);
548  /* Set the LSB mantissa of UART1DIV */
549  UART1->BRR1 |= (uint8_t) BaudRate_Mantissa;

552  UART1->CR2 &= (uint8_t) ~(UART1_CR2_TEN | UART1_CR2_REN);
553  /* Clear the Clock Polarity, lock Phase, Last Bit Clock pulse */
554  UART1->CR3 &= (uint8_t) ~(UART1_CR3_CPOL | UART1_CR3_CPHA | UART1_CR3_LBCL);

560  /* Set the Transmitter Enable bit */
561  UART1->CR2 |= (uint8_t) UART1_CR2_TEN;
562  } else {
563  /* Clear the Transmitter Disable bit */
564  UART1->CR2 &= (uint8_t) (~UART1_CR2_TEN);
565  }
566  if ((uint8_t) (Mode & UART1_MODE_RX_ENABLE)) {
567  /* Set the Receiver Enable bit */
568  UART1->CR2 |= (uint8_t) UART1_CR2_REN;
569  } else {

578  } else {
