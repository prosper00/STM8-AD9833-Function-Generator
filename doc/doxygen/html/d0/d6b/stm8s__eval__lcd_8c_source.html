283 void LCD_RollString(uint8_t Line, uint8_t *ptr, uint16_t speed)
284 {
285 
286  uint8_t CharPos = 0;
287  uint8_t *ptr2;
288 
289  /* Set cursor position at beginning of line */
290  LCD_SendByte(COMMAND_TYPE, Line);
291 
292  ptr2 = ptr;
293 
294  /* Display each character of the string */
295  while (*ptr2 != 0)
296  {
297  if (*ptr != 0)
298  {
299  LCD_SendByte(DATA_TYPE, *ptr);
300  ptr++;
301  }
302  else
303  {
304  LCD_SendByte(DATA_TYPE, ' ');
305  }
306 
307  CharPos++;
308 
309  if (CharPos == LCD_LINE_MAX_CHAR)
310  {
311  _delay_(speed);
312 
313  /* Select the line to be cleared */
314  LCD_SendByte(COMMAND_TYPE, Line);
315 
316  /* Clear the selected line */
317  for (CharPos = 0; CharPos < LCD_LINE_MAX_CHAR; CharPos++)
318  {
319  LCD_SendByte(DATA_TYPE, ' ');
320  }
321  LCD_SendByte(COMMAND_TYPE, Line);
322  CharPos = 0;
323  ptr2++;
324  ptr = ptr2;
325  }
