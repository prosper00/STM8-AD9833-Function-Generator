670  itmask1 = (uint8_t) ((uint8_t) SPI_IT >> (uint8_t) 4);

673  /* Get the SPI_ITPENDINGBIT enable bit status */
674  enablestatus = (uint8_t) ((uint8_t) SPI->SR & itmask2);
675  /* Check the status of the specified SPI interrupt */
676  if (((SPI->ICR & itpos) != RESET) && enablestatus) {
677  /* SPI_ITPENDINGBIT is set */
