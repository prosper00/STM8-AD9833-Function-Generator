00275 TestStatus Buffercmp(uint8_t* pBuffer1, uint8_t* pBuffer2, uint16_t BufferLength)
00276 {
00277   while (BufferLength--)
00278   {
00279     if (*pBuffer1 != *pBuffer2)
00280     {
00281       return FAILED;
00282     }
00283 
00284     pBuffer1++;
00285     pBuffer2++;
00286   }
00287 
00288   return PASSED;
