00073   {
00074     /* Decrement timeout */

00076   }
00077

00079   {
00080     /*If the program counter reaches this section,

00084   }
00085   else
00086   {
00087     /*If the program counter reaches this section, 
00088       the TIM5 is not correctly triggered by TIM6 */
00089     /*Insert break point in the following line */
00090     Result = 0;

00093   while(1)
00094   {

00108   /* Time base configuration */
00109   TIM5_TimeBaseInit(TIM5_PRESCALER_1, 0xFF );
00110
